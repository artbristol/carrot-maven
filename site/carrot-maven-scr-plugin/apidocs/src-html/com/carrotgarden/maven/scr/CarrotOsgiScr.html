008package com.carrotgarden.maven.scr;
009
010import java.io.File;
011import java.util.HashMap;
012import java.util.Map;
013import java.util.Set;
014
015import org.apache.maven.plugin.AbstractMojo;
016import org.apache.maven.project.MavenProject;
017
018import com.carrotgarden.osgi.anno.scr.make.Maker;
019
020/**
021 * Base for Maven goals.
022 */
023public abstract class CarrotOsgiScr extends AbstractMojo {
024
025        /**
026         * Java class extension during class discovery.
027         */
028        protected static final String[] EXTENSIONS = new String[] { "class" };
029
030        /** Find classes from all packages during class discovery */
031        protected static final boolean IS_RECURSIVE = true;
032
033        /**
034         * Exclude java *.class files with this file name regex expression.
035         * 
036         * @required
037         * @parameter default-value= ".*-.*"
038         */
039        protected String excludeFileNameRegex;
040
041        /**
042         * Current maven pom.xml.
043         * 
044         * @readonly
045         * @required
046         * @parameter property="project"
047         */
048        protected MavenProject project;
049
050        /**
051         * Map of "key=value" settings for Eclipse m2e-scr connector plugin.
052         * 
053         * @parameter
054         */
055        protected Map<String, String> eclipseSettings = new HashMap<String, String>();
056
057        /**
058         * Location of generated DS component descriptor files in final bundle.
059         * 
060         * @required
061         * @parameter default-value= "OSGI-INF/service-component"
062         */
063        protected String targetDirectorySCR;
064
065        /**
066         * Default extension used for generated DS component descriptor files.
067         * 
068         * @required
069         * @parameter default-value="xml"
070         */
071        protected String outputExtensionSCR;
072
073        /**
074         * Location of compiled "main" class files.
075         * 
076         * @required
077         * @parameter default-value="${project.build.outputDirectory}"
078         */
079        protected File outputMainClasses;
080
081        /**
082         * Location of compiled "test" class files.
083         * 
084         * @required
085         * @parameter default-value="${project.build.testOutputDirectory}"
086         */
087        protected File outputTestClasses;
088
089        /**
090         * Collection of names of unwanted component service interfaces.
091         * 
092         * @required
093         * @parameter default-value="java.lang.Cloneable"
094         */
095        protected Set<String> excludedServices;
096
097        /**
098         * Should "main" classes be processed?
099         * 
100         * @required
101         * @parameter default-value="true"
102         */
103        protected boolean isProcessMainClasses;
104
105        /**
106         * Should "test" classes be processed?
107         * 
108         * @required
109         * @parameter default-value="false"
110         */
111        protected boolean isProcessTestClasses;
112
113        /**
114         * Should include an empty component descriptor?
115         * 
116         * @required
117         * @parameter default-value="true"
118         */
119        protected boolean isIncludeEmptyDescriptor;
120
121        /**
122         * should generated descriptor resource files be included in final bundle?
123         * 
124         * @required
125         * @parameter default-value="true"
126         */
127        protected boolean isIncludeGeneratedDescritors;
128
129        /**
130         * Collection of names of known maven project packaging types for which to
131         * invoke this plug-in.
132         * 
133         * @required
134         * @parameter default-value="bundle"
135         */
136        protected Set<String> properPackaging;
137
138        // ####################################################
139
140        /**
141         * Build output directory, such as "./target/OSGI-INF/service-component".
142         */
143        protected File outputDirectorySCR() {
144
145                return new File(outputMainClasses, targetDirectorySCR);
146
147        }
148
149        /**
150         * Check if current project packaging is proper for goal execution.
151         */
152        protected boolean isProperPackaging() {
153
154                final String packaging = project.getPackaging();
155
156                if (properPackaging.contains(packaging)) {
157                        getLog().debug("proper packaging=" + packaging);
158                        return true;
159                } else {
160                        getLog().debug("proper packaging=" + packaging);
161                        return false;
162                }
163
164        }
165
166        private Maker maker;
167
168        /**
169         * Maker of DS component descriptors.
170         */
171        protected Maker maker() {
172                if (maker == null) {
173                        maker = new Maker(excludedServices);
174                }
175                return maker;
176        }
177
178}
