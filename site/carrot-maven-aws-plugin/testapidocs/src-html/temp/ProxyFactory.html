008package temp;
009
010import java.lang.reflect.InvocationHandler;
011import java.lang.reflect.InvocationTargetException;
012import java.lang.reflect.Method;
013import java.lang.reflect.Proxy;
014import java.util.Arrays;
015import java.util.HashMap;
016import java.util.Map;
017
018public class ProxyFactory {
019
020        @SuppressWarnings("unchecked")
021        public static <T> T makeProxy(final Class<T> face, final Object instance,
022                        final Object instanceProxy) {
023
024                return (T) Proxy.newProxyInstance( //
025
026                                face.getClassLoader(), //
027
028                                new Class[] { face }, //
029
030                                new InvocationHandler() {
031
032                                        private final Map<MethodEntry, Method> //
033                                        methodProxyMap = new HashMap<MethodEntry, Method>();
034
035                                        /** find all methods in adapter object */
036
037                                        {
038
039                                                final Method[] methodArray = instanceProxy.getClass()
040                                                                .getDeclaredMethods();
041
042                                                for (final Method method : methodArray) {
043                                                        methodProxyMap.put(new MethodEntry(method), method);
044                                                }
045
046                                        }
047
048                                        @Override
049                                        public Object invoke( //
050                                                        final Object proxy, //
051                                                        final Method method, //
052                                                        final Object[] args //
053                                        ) throws Throwable {
054
055                                                try {
056
057                                                        final Method methodProxy = methodProxyMap
058                                                                        .get(new MethodEntry(method));
059
060                                                        if (methodProxy == null) {
061                                                                return method.invoke(instance, args);
062                                                        } else {
063                                                                return methodProxy.invoke(instanceProxy, args);
064                                                        }
065
066                                                } catch (final InvocationTargetException e) {
067                                                        throw e.getTargetException();
068                                                }
069                                        }
070
071                                }
072
073                );
074
075        }
076
077        private static class MethodEntry {
078
079                private final String methodName;
080                private final Class<?>[] parameterTypes;
081
082                public MethodEntry(final Method m) {
083                        methodName = m.getName();
084                        parameterTypes = m.getParameterTypes();
085                }
086
087                @Override
088                public boolean equals(final Object other) {
089
090                        final MethodEntry entry = (MethodEntry) other;
091
092                        return methodName.equals(entry.methodName)
093                                        && Arrays.equals(parameterTypes, entry.parameterTypes);
094
095                }
096
097                @Override
098                public int hashCode() {
099
100                        return methodName.hashCode();
101
102                }
103
104        }
105
106}
