008package com.carrotgarden.maven.aws;
009
010import org.apache.maven.execution.MavenSession;
011import org.apache.maven.plugin.AbstractMojo;
012import org.apache.maven.project.MavenProject;
013import org.apache.maven.settings.Settings;
014import org.slf4j.Logger;
015import org.slf4j.impl.MavenLoggerFactory;
016
017/**
018 * base plug-in mojo
019 */
020public abstract class CarrotMojo extends AbstractMojo {
021
022        /**
023         * default amazon tag for resource names
024         * 
025         * @parameter default-value="Name"
026         */
027        private String amazonTagName;
028
029        protected String amazonTagName() {
030                return amazonTagName;
031        }
032
033        /** do not use during class init */
034        protected Logger getLogger(final Class<?> klaz) {
035                return MavenLoggerFactory.getLogger(klaz, getLog());
036        }
037
038        /**
039         * The Maven Session *
040         * 
041         * @required
042         * @readonly
043         * @parameter expression="${session}"
044         */
045        private MavenSession session;
046
047        protected MavenSession session() {
048                return session;
049        }
050
051        /**
052         * @readonly
053         * @required
054         * @parameter expression="${project}"
055         */
056        private MavenProject project;
057
058        protected MavenProject project() {
059                return project;
060        }
061
062        /**
063         * @readonly
064         * @required
065         * @parameter expression="${settings}"
066         */
067        private Settings settings;
068
069        protected Settings settings() {
070                return settings;
071        }
072
073        /**
074         * AWS
075         * 
076         * <a href= "http://docs.amazonwebservices.com/general/latest/gr/rande.html"
077         * >region name, such as us-east-1,</a>
078         * 
079         * which controls amazon region selection;
080         * 
081         * alternatively, see {@link #amazonRegionProperty}
082         * 
083         * @required
084         * @parameter default-value="us-east-1"
085         */
086        private String amazonRegion;
087
088        /**
089         * name of project.property which, if set dynamically, will be used instead
090         * of plugin property {@link #amazonRegion}
091         * 
092         * @parameter
093         */
094        private String amazonRegionProperty;
095
096        /** prefer project.properies over plug-in property */
097        protected String amazonRegion() {
098                return projectValue(amazonRegion, amazonRegionProperty);
099        }
100
101        /**
102         * @return value, if present; else substitute {@link #amazonRegion()} into
103         *         format expression
104         */
105        protected String amazonEndpoint(final String value, final String format) {
106                if (value == null) {
107                        return String.format(format, amazonRegion());
108                } else {
109                        return value;
110                }
111        }
112
113        protected String projectValue(final String propValue, final String propName) {
114                if (propName == null) {
115                        return propValue;
116                } else {
117                        return (String) project().getProperties().get(propName);
118                }
119
120        }
121
122}
