008package com.carrotgarden.maven.aws.ecc;
009
010import java.io.File;
011import java.util.Properties;
012
013import org.apache.maven.plugin.MojoFailureException;
014
015import com.amazonaws.services.ec2.model.BlockDeviceMapping;
016import com.amazonaws.services.ec2.model.EbsBlockDevice;
017import com.amazonaws.services.ec2.model.Image;
018import com.carrotgarden.maven.aws.ecc.ElasticCompute.State;
019import com.carrotgarden.maven.aws.util.Util;
020
021/**
022 * register/create new ami image from existing AWS ElasticCompute instance
023 * 
024 * @goal elastic-compute-image-register
025 * 
026 * @phase prepare-package
027 * 
028 * @inheritByDefault true
029 * 
030 * @requiresDependencyResolution test
031 * 
032 */
033public class CarrotAwsElasCompImageReg extends CarrotAwsElasComp {
034
035        /**
036         * AWS ElasticCompute <b>instance id</b> to use for image create; if
037         * provided explicitly in pom.xml, then it will override value form
038         * properties file; this value must come either from image properties file
039         * of from pom.xml properties
040         * 
041         * @parameter
042         */
043        protected String imageInstanceId;
044
045        /**
046         * AWS ElasticCompute ami image name; must be unique under your aws account;
047         * 
048         * @required
049         * @parameter default-value="amazon-image-name"
050         */
051        protected String imageName;
052
053        /**
054         * AWS ElasticCompute ami image description
055         * 
056         * @required
057         * @parameter default-value="amazon-image-description"
058         */
059        protected String imageDescription;
060
061        /**
062         * AWS ElasticCompute image create input properties file; normally created
063         * by previous cloud formation invocation; must include InstanceId property;
064         * 
065         * @required
066         * @parameter default-value="./target/formation/formation-output.properties"
067         */
068        protected File imagePropertiesInputFile;
069
070        /**
071         * name of the property in input properties file that will be used to
072         * discover <b>InstanceId</b> to use as basis for image create operation
073         * 
074         * @required
075         * @parameter default-value="InstanceId"
076         */
077        protected String imagePropertyInstanceId;
078
079        @Override
080        public void execute() throws MojoFailureException {
081
082                try {
083
084                        getLog().info("image reg init [" + imageName + "]");
085
086                        final Properties props = Util.propsLoad(getLog(), imagePropertiesInputFile);
087
088                        final String instanceId;
089                        if (imageInstanceId == null) {
090                                instanceId = props.getProperty(imagePropertyInstanceId);
091                        } else {
092                                instanceId = imageInstanceId;
093                        }
094
095                        final ElasticCompute compute = newElasticCompute();
096
097                        final Image image = compute.imageRegister( //
098                                        instanceId, imageName, imageDescription);
099
100                        final State state = State.fromValue(image.getState());
101
102                        switch (state) {
103                        case AVAILABLE:
104                                break;
105                        default:
106                                throw new IllegalStateException("image reg failed : \n" + image);
107                        }
108
109                        final String imageId = image.getImageId();
110
111                        /** tag image */
112                        compute.tagCreate(imageId, "Name", imageName);
113
114                        /** tag image devices */
115                        for (final BlockDeviceMapping blockDevice : image
116                                        .getBlockDeviceMappings()) {
117
118                                final EbsBlockDevice elasticDevice = blockDevice.getEbs();
119
120                                if (elasticDevice == null) {
121                                        continue;
122                                }
123
124                                final String snapshotId = elasticDevice.getSnapshotId();
125
126                                if (snapshotId == null) {
127                                        continue;
128                                }
129
130                                compute.tagCreate(snapshotId, "Name", imageName);
131
132                        }
133
134                        getLog().info("stack reg image=\n" + image);
135
136                        getLog().info("image reg done [" + imageName + "]");
137
138                } catch (final Exception e) {
139
140                        throw new MojoFailureException("bada-boom", e);
141
142                }
143
144        }
145
146}
