008package com.carrotgarden.maven.aws.grv;
009
010import groovy.lang.Binding;
011import groovy.lang.GroovyShell;
012
013import java.io.File;
014
015import org.apache.maven.project.MavenProject;
016
017/**
018 * groovy script executor which exposes maven project
019 */
020public class CarrotGroovyRunner {
021
022        private final MavenProject project;
023
024        public CarrotGroovyRunner(final MavenProject project) {
025                this.project = project;
026        }
027
028        public Binding binding() {
029
030                final Binding binding = new Binding();
031
032                binding.setProperty("project", project);
033
034                return binding;
035
036        }
037
038        public Object execute(final File script) throws Exception {
039
040                final GroovyShell shell = new GroovyShell(binding());
041
042                final Object result = shell.evaluate(script);
043
044                return result;
045
046        }
047
048        public Object execute(final String script) throws Exception {
049
050                final GroovyShell shell = new GroovyShell(binding());
051
052                final Object result = shell.evaluate(script);
053
054                return result;
055
056        }
057
058}
