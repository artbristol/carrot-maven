008package com.carrotgarden.maven.aws.dns;
009
010import java.util.List;
011
012import org.apache.maven.plugin.MojoExecutionException;
013import org.apache.maven.plugin.MojoFailureException;
014
015import com.carrotgarden.maven.aws.util.Util;
016
017/**
018 * route53:
019 * 
020 * <b><a href=
021 * "http://docs.amazonwebservices.com/Route53/latest/DeveloperGuide/ListingRRS.html"
022 * >list dns zone entires</a></b>
023 * 
024 * @goal route53-list-zone
025 * 
026 * @phase prepare-package
027 * 
028 * @inheritByDefault true
029 * 
030 * @requiresDependencyResolution test
031 * 
032 */
033public class CarrotAwsNameServListZone extends CarrotAwsNameServ {
034
035        /**
036         * dns zone name to be listed
037         * 
038         * @required
039         * @parameter default-value="default.example.com"
040         */
041        protected String dnsZoneName;
042
043        /**
044         * name of the maven project.property that will contain dns name list after
045         * execution of this maven goal
046         * 
047         * @required
048         * @parameter default-value="dnsNameList"
049         */
050        protected String dnsResultProperty;
051
052        /**
053         * separator character to use when building dns result name list
054         * 
055         * @required
056         * @parameter default-value=";"
057         */
058        protected String dnsResultSeparator;
059
060        @Override
061        public void execute() throws MojoExecutionException, MojoFailureException {
062
063                try {
064
065                        getLog().info("dns list init [" + dnsZoneName + "]");
066
067                        final Route53 route53 = newRoute53();
068
069                        final List<String> list = route53.listZone(dnsZoneName);
070
071                        final String dnsResultNameList = Util.concatenate(list,
072                                        dnsResultSeparator);
073
074                        getLog().info("dns name list : " + dnsResultNameList);
075
076                        project().getProperties().put(dnsResultProperty, dnsResultNameList);
077
078                        getLog().info("dns list done [" + dnsZoneName + "]");
079
080                } catch (final Exception e) {
081
082                        throw new MojoFailureException("bada-boom", e);
083
084                }
085
086        }
087
088}
