008package com.carrotgarden.maven.aws;
009
010import org.apache.maven.execution.MavenSession;
011import org.apache.maven.plugin.AbstractMojo;
012import org.apache.maven.project.MavenProject;
013import org.apache.maven.settings.Settings;
014import org.slf4j.Logger;
015import org.slf4j.impl.MavenLoggerFactory;
016
017/**
018 */
019public abstract class CarrotAws extends AbstractMojo {
020
021        /** do not use during class init */
022        protected Logger getLogger(final Class<?> klaz) {
023                return MavenLoggerFactory.getLogger(klaz, getLog());
024        }
025
026        /**
027         * The Maven Session *
028         * 
029         * @required
030         * @readonly
031         * @parameter expression="${session}"
032         */
033        private MavenSession session;
034
035        protected MavenSession session() {
036                return session;
037        }
038
039        /**
040         * @readonly
041         * @required
042         * @parameter expression="${project}"
043         */
044        private MavenProject project;
045
046        protected MavenProject project() {
047                return project;
048        }
049
050        /**
051         * @readonly
052         * @required
053         * @parameter expression="${settings}"
054         */
055        private Settings settings;
056
057        protected Settings settings() {
058                return settings;
059        }
060
061        /**
062         * AWS
063         * 
064         * <a href= "http://docs.amazonwebservices.com/general/latest/gr/rande.html"
065         * >region name, such as us-east-1,</a>
066         * 
067         * which controls amazon region selection;
068         * 
069         * @required
070         * @parameter default-value="us-east-1"
071         */
072        private String amazonRegion;
073
074        /**
075         * name of project.property which, if set dynamically, will be used instead
076         * of plugin property {@link #amazonRegion}
077         * 
078         * @parameter
079         */
080        private String amazonRegionProperty;
081
082        /** prefer project.properies over plug-in property */
083        protected String amazonRegion() {
084                if (amazonRegionProperty == null) {
085                        return amazonRegion;
086                } else {
087                        return (String) project().getProperties().get(amazonRegionProperty);
088                }
089        }
090
091}
