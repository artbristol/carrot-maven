008package com.carrotgarden.maven.aws.cfn;
009
010import org.apache.maven.plugin.MojoFailureException;
011
012/**
013 * cloud formation:
014 * 
015 * <b><a href=
016 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/APIReference/API_UpdateStack.html"
017 * >update stack</a></b>
018 * 
019 * based on:
020 * 
021 * <b>stack name</b>
022 * 
023 * ; wait for completion or fail ({@link #stackTimeout})
024 * 
025 * @goal cloud-formation-update
026 * 
027 * @phase prepare-package
028 * 
029 * @inheritByDefault true
030 * 
031 * @requiresDependencyResolution test
032 * 
033 */
034public class CloudFormUpdateStack extends CloudForm {
035
036        /**
037         * {@inheritDoc}
038         */
039        @Override
040        public void execute() throws MojoFailureException {
041
042                try {
043
044                        getLog().info("stack update init [" + stackName() + "]");
045
046                        getLog().error("stack update TODO");
047
048                        getLog().info("stack update done [" + stackName() + "]");
049
050                } catch (final Exception e) {
051
052                        throw new MojoFailureException("bada-boom", e);
053
054                }
055
056        }
057
058}
