008package com.carrotgarden.maven.aws.ssh;
009
010/**
011 */
012
013import java.io.File;
014import java.io.FileReader;
015import java.io.IOException;
016import java.util.Properties;
017import java.util.Set;
018
019import org.slf4j.Logger;
020
021import com.carrotgarden.maven.aws.CarrotAws;
022import com.carrotgarden.maven.aws.cfn.CloudFormation;
023
024/**
025 * 
026 */
027public abstract class CarrotAwsSecuShel extends CarrotAws {
028
029        /**
030         * ssh key file
031         * 
032         * @required
033         * @parameter default-value="${user.home}/.amazon/ssh-key.pem"
034         */
035        protected File sshKeyFile;
036
037        /**
038         * ssh user name
039         * 
040         * @required
041         * @parameter default-value="ubuntu"
042         */
043        protected String sshUser;
044
045        /**
046         * ssh host name
047         * 
048         * @required
049         * @parameter default-value="builder.example.com"
050         */
051        protected String sshHost;
052
053        /**
054         * ssh run expected exit success status collection; such as
055         * 
056         * @required
057         * @parameter
058         */
059        protected Set<Integer> sshStatusSuccess;
060
061        /**
062         * AWS CloudFormation stack create execution result
063         * 
064         * <a href=
065         * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/concept-outputs.html"
066         * >Outputs Section</a>
067         * 
068         * output properties file - used to substitute ssHost
069         * 
070         * @required
071         * @parameter default-value="./target/formation/formation-output.properties"
072         */
073        protected File stackPropertiesOutputFile;
074
075        /**
076         * How many times to attempt to retry ssh connection before giving up
077         * 
078         * @required
079         * @parameter default-value="5"
080         */
081        protected int sshConnectRetries;
082
083        /**
084         * How long (in seconds) to wait if a ssh connection fails before retrying
085         * 
086         * @required
087         * @parameter default-value="10"
088         */
089        protected int sshConnectTimeout;
090
091        //
092
093        protected boolean isStatusSuccess(final Integer status) {
094                return sshStatusSuccess.contains(status);
095        }
096
097        protected void ensureStatusSuccess(final Integer status) throws Exception {
098
099                if (isStatusSuccess(status)) {
100                        return;
101                }
102
103                throw new IllegalStateException("invalid ssh exit status = " + status);
104
105        }
106
107        protected SecureShell getSecureShell() throws Exception {
108
109                final Logger logger = getLogger(CloudFormation.class);
110
111                final String sshHostConverted = getSSHHost();
112
113                final SecureShell ssh = new SecureShell(logger, sshKeyFile, sshUser,
114                                sshHostConverted, sshConnectRetries, sshConnectTimeout);
115
116                return ssh;
117
118        }
119
120        private String getSSHHost() {
121                if (stackPropertiesOutputFile.exists()) {
122                        getLog().info("Attempt to property substitute sshHost " + sshHost);
123                        final Properties props = new Properties();
124
125                        try {
126                                final FileReader fr = new FileReader(stackPropertiesOutputFile);
127                                props.load(fr);
128                        } catch (final IOException e) {
129                                // Ignore exception
130                                getLog().warn(
131                                                "Failed to read properties from "
132                                                                + stackPropertiesOutputFile);
133                                return sshHost;
134                        }
135
136                        final String sshHostConverted = props.getProperty(sshHost, sshHost);
137                        getLog().debug("Return sshHost as " + sshHostConverted);
138                        return sshHostConverted;
139                }
140                return sshHost;
141        }
142
143}
