009package com.carrotgarden.maven.aws.ecc;
010
011import org.apache.maven.plugin.MojoFailureException;
012
013/**
014 * delete existing ami image with all ebs snapshots
015 * 
016 * @goal elastic-compute-image-delete
017 * 
018 * @phase prepare-package
019 * 
020 * @inheritByDefault true
021 * 
022 * @requiresDependencyResolution test
023 * 
024 */
025public class ElastiCompImageDelete extends ElastiComp {
026
027        /**
028         * AWS ElasticCompute existing image id to delete; also can loaded via
029         * {@link #imageIdProperty}
030         * 
031         * @required
032         * @parameter default-value="ami-12345678"
033         */
034        private String imageId;
035
036        /**
037         * name of project.property which, if set dynamically, will be used instead
038         * of static plug-in property {@link #imageId}
039         * 
040         * @parameter
041         */
042        private String imageIdProperty;
043
044        protected String imageId() throws Exception {
045                return projectValue(imageId, imageIdProperty);
046        }
047
048        @Override
049        public void execute() throws MojoFailureException {
050
051                try {
052
053                        getLog().info("image delete init : " + imageId());
054
055                        final CarrotElasticCompute compute = newElasticCompute();
056
057                        compute.imageDelete(imageId());
058
059                        getLog().info("image delete done : " + imageId());
060
061                } catch (final Exception e) {
062
063                        throw new MojoFailureException("bada-boom", e);
064
065                }
066
067        }
068
069}
