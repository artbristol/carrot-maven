008package com.carrotgarden.maven.aws.cfn;
009
010import java.io.File;
011import java.util.HashMap;
012import java.util.List;
013import java.util.Map;
014import java.util.Properties;
015
016import org.apache.maven.plugin.MojoFailureException;
017
018import com.amazonaws.services.cloudformation.model.Output;
019import com.amazonaws.services.cloudformation.model.Stack;
020import com.amazonaws.services.cloudformation.model.StackStatus;
021import com.carrotgarden.maven.aws.util.Util;
022
023/**
024 * cloud formation:
025 * 
026 * <b><a href=
027 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/APIReference/API_CreateStack.html"
028 * >create stack</a></b>
029 * 
030 * based on:
031 * 
032 * <b>stack name</b>,
033 * 
034 * <b><a href=
035 * "http://aws.amazon.com/cloudformation/aws-cloudformation-templates" >stack
036 * template</a></b>
037 * 
038 * ({@link #stackTemplateFile}),
039 * 
040 * <b><a href=
041 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/parameters-section-structure.html"
042 * >stack parameters</a></b>
043 * 
044 * ({@link #stackPropertiesInputFile} + {@link #stackInputParams}),
045 * 
046 * and produce a
047 * 
048 * <b><a href=
049 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/concept-outputs.html"
050 * >stack output</a></b>
051 * 
052 * ({@link #stackPropertiesOutputFile})
053 * 
054 * as well as inject output into into project.properties
055 * 
056 * ; wait for completion or fail ({@link #stackTimeout});
057 * 
058 * @goal cloud-formation-create
059 * 
060 * @phase prepare-package
061 * 
062 * @inheritByDefault true
063 * 
064 * @requiresDependencyResolution test
065 * 
066 */
067public class CarrotAwsCloudFormCreate extends CarrotAwsCloudForm {
068
069        /**
070         * AWS CloudFormation
071         * 
072         * <a href=
073         * "http://aws.amazon.com/cloudformation/aws-cloudformation-templates"
074         * >template</a>
075         * 
076         * file
077         * 
078         * @required
079         * @parameter default-value="./target/formation/formation.template"
080         */
081        protected File stackTemplateFile;
082
083        /**
084         * AWS CloudFormation
085         * 
086         * <a href=
087         * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/parameters-section-structure.html"
088         * >Parameters Declaration</a>
089         * 
090         * stack template input parameters; optional; overrides settings from
091         * #stackPropertiesInputFile
092         * 
093         * @parameter
094         */
095        protected Map<String, String> stackInputParams = new HashMap<String, String>();
096
097        /**
098         * AWS CloudFormation stack create execution
099         * 
100         * <a href=
101         * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/parameters-section-structure.html"
102         * >Parameters Declaration</a>
103         * 
104         * input properties file;
105         * 
106         * will be overridden by #stackInputParams if any
107         * 
108         * @required
109         * @parameter default-value="./target/formation/formation-input.properties"
110         */
111        protected File stackPropertiesInputFile;
112
113        /**
114         * AWS CloudFormation stack create execution result
115         * 
116         * <a href=
117         * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/UserGuide/concept-outputs.html"
118         * >Outputs Section</a>
119         * 
120         * output properties file
121         * 
122         * @required
123         * @parameter default-value="./target/formation/formation-output.properties"
124         */
125        protected File stackPropertiesOutputFile;
126
127        /**
128         * should inject stack operation output into the project.properties?
129         * 
130         * @required
131         * @parameter default-value="true"
132         */
133        protected boolean stackIsInjectOutputProperties;
134
135        /**
136         * should persist stack operation output into
137         * {@link #stackPropertiesOutputFile}
138         * 
139         * @required
140         * @parameter default-value="true"
141         */
142        protected boolean stackIsPersistOutputProperties;
143
144        /**
145         * {@inheritDoc}
146         */
147        @Override
148        public void execute() throws MojoFailureException {
149
150                try {
151
152                        getLog().info("stack create init [" + stackName() + "]");
153
154                        final Properties stackInputProps = Util.propsLoad(getLog(),
155                                        stackPropertiesInputFile);
156
157                        final Map<String, String> pluginProps = mergePluginProps(
158                                        stackInputProps, stackInputParams);
159
160                        final Map<String, String> stackTemplateParams = loadTemplateParameters(
161                                        stackTemplateFile, pluginProps);
162
163                        final CloudFormation formation = newCloudFormation(
164                                        stackTemplateFile, stackTemplateParams);
165
166                        formation.logParamList();
167
168                        final Stack stack = formation.stackCreate();
169
170                        final StackStatus status = StackStatus.fromValue(stack
171                                        .getStackStatus());
172
173                        switch (status) {
174                        case CREATE_COMPLETE:
175                                break;
176                        default:
177                                throw new IllegalStateException("stack create failed");
178                        }
179
180                        //
181
182                        getLog().info("stack create stack=\n" + stack);
183
184                        getLog().info("stack create output:");
185
186                        final List<Output> outputList = stack.getOutputs();
187
188                        final Properties outputProps = new Properties();
189
190                        for (final Output output : outputList) {
191
192                                final String key = output.getOutputKey();
193                                final String value = output.getOutputValue();
194
195                                outputProps.put(key, value);
196
197                                getLog().info("\t" + key + "=" + value);
198
199                        }
200
201                        if (stackIsInjectOutputProperties) {
202
203                                project().getProperties().putAll(outputProps);
204
205                                getLog().info(
206                                                "stack create output is injected in project.properties]");
207
208                        }
209
210                        if (stackIsPersistOutputProperties) {
211
212                                Util.propsSave(getLog(), outputProps, stackPropertiesOutputFile);
213
214                                getLog().info(
215                                                "stack create output is persisted to : "
216                                                                + stackPropertiesOutputFile);
217
218                        }
219
220                        //
221
222                        getLog().info("stack create done [" + stackName() + "]");
223
224                } catch (final Exception e) {
225
226                        throw new MojoFailureException("bada-boom", e);
227
228                }
229
230        }
231
232}
