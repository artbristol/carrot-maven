039    public class CloudFormation {
040    
041            private final Logger logger;
042    
043            private final String name;
044            private final String template;
045            private final List<Parameter> paramList;
046    
047            private final long timeout;
048    
049            private final String awsAccessKey;
050            private final String awsSecretKey;
051    
052            private final AmazonCloudFormation amazonClient;
053    
054            private final long waitBetweenAttempts;
055    
056            public CloudFormation(final Logger logger, final String stackName,
057                            final String stackTemplate, final Map<String, String> stackParams,
058                            final long timeout, final String awsAccessKey,
059                            final String awsSecretKey) {
060    
061                    this.logger = logger;
062    
063                    this.name = stackName;
064                    this.template = stackTemplate;
065                    this.paramList = convert(stackParams);
066    
067                    this.awsAccessKey = awsAccessKey;
068                    this.awsSecretKey = awsSecretKey;
069    
070                    this.timeout = timeout;
071    
072                    this.amazonClient = getClient();
073    
074                    this.waitBetweenAttempts = 10; // query every 10s
075    
076            }
077    
078            private List<Parameter> convert(final Map<String, String> paraMap) {
079    
080                    final List<Parameter> list = Lists.newArrayList();
081    
082                    if (paraMap == null || paraMap.values().size() == 0) {
083                            return list;
084                    }
085    
086                    for (final String name : paraMap.keySet()) {
087                            final Parameter parameter = new Parameter();
088                            parameter.setParameterKey(name);
089                            parameter.setParameterValue(paraMap.get(name));
090                            list.add(parameter);
091                    }
092    
093                    return list;
094    
095            }
096    
097            /**
098             */
099            public Stack stackDelete() throws Exception {
100    
101                    final DeleteStackRequest request = new DeleteStackRequest();
102    
103                    request.withStackName(name);
104    
105                    amazonClient.deleteStack(request);
106    
107                    final Stack stack = waitForStackDelete();
108    
109                    return stack;
110    
111            }
112    
113            /**
114             */
115            public Stack stackCreate() throws Exception {
116    
117                    final CreateStackRequest request = new CreateStackRequest();
118    
119                    request.withStackName(name);
120                    request.withParameters(paramList);
121                    request.withTemplateBody(template);
122    
123                    amazonClient.createStack(request);
124    
125                    final Stack stack = waitForStackCreate();
126    
127                    return stack;
128    
129            }
130    
131            private AmazonCloudFormation getClient() {
132    
133                    final AWSCredentials credentials = new BasicAWSCredentials(
134                                    this.awsAccessKey, this.awsSecretKey);
135    
136                    final AmazonCloudFormation amazonClient = new AmazonCloudFormationAsyncClient(
137                                    credentials);
138    
139                    return amazonClient;
140    
141            }
142    
143            private boolean isStackValid(final Stack stack) {
144                    return stack != null;
145            }
146    
147            private Stack waitForStackDelete() throws Exception {
148    
149                    final long timeStart = System.currentTimeMillis();
150    
151                    while (true) {
152    
153                            if (isTimeoutPending(timeStart)) {
154                                    return newStackWithStatus(StackStatus.DELETE_FAILED,
155                                                    "stack delete timeout");
156                            }
157    
158                            Stack stack = null;
159                            try {
160                                    stack = getStack();
161                            } catch (final Exception e) {
162                                    return newStackWithStatus(StackStatus.DELETE_FAILED,
163                                                    e.toString());
164                            }
165    
166                            if (!isStackValid(stack)) {
167                                    return newStackWithStatus(StackStatus.DELETE_COMPLETE,
168                                                    "stack delete invalid/missing");
169                            }
170    
171                            final StackStatus status = StackStatus.fromValue(stack
172                                            .getStackStatus());
173    
174                            switch (status) {
175                            case DELETE_IN_PROGRESS:
176                                    final long timeCurrent = System.currentTimeMillis();
177                                    final long timeDiff = timeCurrent - timeStart;
178                                    logger.info("stack delete in progress; time=" + timeDiff / 1000);
179                                    sleep();
180                                    continue;
181                            case DELETE_COMPLETE:
182                                    logger.info("stack delete complete");
183                                    printStackEvents();
184                                    return stack;
185                            default:
186                                    logger.error("stack delete failed");
187                                    return stack;
188                            }
189    
190                    }
191    
192            }
193    
194            private Stack newStackWithStatus(final StackStatus status,
195                            final String reason) {
196    
197                    final Stack stack = new Stack();
198    
199                    stack.setStackName(name);
200                    stack.setStackStatus(status);
201                    stack.setStackStatusReason(reason);
202    
203                    return stack;
204    
205            }
206    
207            private Stack waitForStackCreate() throws Exception {
208    
209                    final long timeStart = System.currentTimeMillis();
210    
211                    while (true) {
212    
213                            if (isTimeoutPending(timeStart)) {
214                                    return newStackWithStatus(StackStatus.CREATE_FAILED,
215                                                    "stack create timeout");
216                            }
217    
218                            Stack stack = null;
219                            try {
220                                    stack = getStack();
221                            } catch (final Exception e) {
222                                    return newStackWithStatus(StackStatus.CREATE_FAILED,
223                                                    e.toString());
224                            }
225    
226                            if (!isStackValid(stack)) {
227                                    return newStackWithStatus(StackStatus.CREATE_FAILED,
228                                                    "stack create invalid/missing");
229                            }
230    
231                            final StackStatus status = StackStatus.fromValue(stack
232                                            .getStackStatus());
233    
234                            switch (status) {
235                            case CREATE_IN_PROGRESS:
236                                    final long timeCurrent = System.currentTimeMillis();
237                                    final long timeDiff = timeCurrent - timeStart;
238                                    logger.info("stack create in progress; time=" + timeDiff / 1000);
239                                    sleep();
240                                    continue;
241                            case CREATE_COMPLETE:
242                                    logger.info("stack create success");
243                                    printStackEvents();
244                                    return stack;
245                            default:
246                                    logger.error("stack create failure");
247                                    return stack;
248                            }
249    
250                    }
251    
252            }
253    
254            private void printStackEvents() {
255    
256                    final DescribeStackEventsRequest request = new DescribeStackEventsRequest();
257    
258                    request.withStackName(name);
259    
260                    final DescribeStackEventsResult describeStackEvents = amazonClient
261                                    .describeStackEvents(request);
262    
263                    final List<StackEvent> stackEvents = describeStackEvents
264                                    .getStackEvents();
265    
266                    Collections.reverse(stackEvents);
267    
268                    logger.info("stack events:");
269    
270                    for (final StackEvent event : stackEvents) {
271    
272                            final StringBuilder text = new StringBuilder(128);
273    
274                            text.append("\n\t");
275                            text.append("time=");
276                            text.append(event.getTimestamp());
277    
278                            text.append("\n\t");
279                            text.append("id=");
280                            text.append(event.getEventId());
281    
282                            text.append("\n\t");
283                            text.append("type=");
284                            text.append(event.getResourceType());
285    
286                            text.append("\n\t");
287                            text.append("status=");
288                            text.append(event.getResourceStatus());
289    
290                            text.append("\n\t");
291                            text.append("reason=");
292                            text.append(event.getResourceStatusReason());
293    
294                            logger.info("event {}", text);
295    
296                    }
297    
298            }
299    
300            private boolean isTimeoutPending(final long startTime) {
301                    return (System.currentTimeMillis() - startTime) > (timeout * 1000);
302            }
303    
304            private Stack getStack() throws Exception {
305    
306                    final DescribeStacksRequest request = new DescribeStacksRequest();
307    
308                    final DescribeStacksResult result = amazonClient
309                                    .describeStacks(request);
310    
311                    for (final Stack stack : result.getStacks()) {
312                            if (name.equals(stack.getStackName())) {
313                                    return stack;
314                            }
315                    }
316    
317                    return null;
318    
319            }
320    
321            private void sleep() throws Exception {
322                    try {
323                            Thread.sleep(waitBetweenAttempts * 1000);
324                    } catch (final InterruptedException ie) {
325                            throw new IllegalStateException("operation interrupted; "
326                                            + "resources are left in inconsistent state; "
327                                            + "requires manual intervention");
328                    }
329            }
