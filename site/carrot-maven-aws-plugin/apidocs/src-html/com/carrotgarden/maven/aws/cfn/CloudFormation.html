010import java.util.Collections;
011import java.util.List;
012import java.util.Map;

014import org.slf4j.Logger;

016import com.amazonaws.auth.AWSCredentials;
017import com.amazonaws.auth.BasicAWSCredentials;

019import com.amazonaws.services.cloudformation.AmazonCloudFormationAsyncClient;
020import com.amazonaws.services.cloudformation.model.CreateStackRequest;
021import com.amazonaws.services.cloudformation.model.DeleteStackRequest;

023import com.amazonaws.services.cloudformation.model.DescribeStackEventsResult;
024import com.amazonaws.services.cloudformation.model.DescribeStacksRequest;

039public class CloudFormation {
040
041        private final Logger logger;
042
043        private final String name;
044        private final String template;
045        private final List<Parameter> paramList;
046
047        private final long timeout;
048
049        private final String awsAccessKey;
050        private final String awsSecretKey;
051
052        private final AmazonCloudFormation amazonClient;
053
054        private final long waitBetweenAttempts;
055
056        private final String endpoint;
057
058        public CloudFormation(final Logger logger, final String stackName,
059                        final String stackTemplate, final Map<String, String> stackParams,
060                        final long timeout, final String awsAccessKey,
061                        final String awsSecretKey, final String endpoint) {
062
063                this.logger = logger;
064
065                this.name = stackName;
066                this.template = stackTemplate;
067                this.paramList = convert(stackParams);
068
069                this.awsAccessKey = awsAccessKey;
070                this.awsSecretKey = awsSecretKey;
071
072                this.timeout = timeout;
073
074                this.endpoint = endpoint;
075
076                this.waitBetweenAttempts = 10; // query every 10s
077
078                this.amazonClient = newClient(); // keep last
079
080        }
081
082        private List<Parameter> convert(final Map<String, String> paraMap) {
083
084                final List<Parameter> list = Lists.newArrayList();
085
086                if (paraMap == null || paraMap.values().size() == 0) {
087                        return list;
088                }
089
090                for (final String name : paraMap.keySet()) {
091                        final Parameter parameter = new Parameter();
092                        parameter.setParameterKey(name);
093                        parameter.setParameterValue(paraMap.get(name));
094                        list.add(parameter);
095                }
096
097                return list;
098
099        }
100
101        /**
102         */
103        public Stack stackDelete() throws Exception {
104
105                final DeleteStackRequest request = new DeleteStackRequest();
106
107                request.withStackName(name);
108
109                amazonClient.deleteStack(request);
110
111                final Stack stack = waitForStackDelete();
112
113                return stack;
114
115        }
116
117        /**
118         */
119        public Stack stackCreate() throws Exception {
120
121                final CreateStackRequest request = new CreateStackRequest();
122
123                request.withStackName(name);
124                request.withParameters(paramList);
125                request.withTemplateBody(template);
126
127                amazonClient.createStack(request);
128
129                final Stack stack = waitForStackCreate();
130
131                return stack;
132
133        }
134
135        private AmazonCloudFormation newClient() {
136
137                final AWSCredentials credentials = new BasicAWSCredentials(
138                                this.awsAccessKey, this.awsSecretKey);
139
140                final AmazonCloudFormation amazonClient = new AmazonCloudFormationAsyncClient(
141                                credentials);
142
143                logger.info("stack endpoint : {}", endpoint);
144
145                amazonClient.setEndpoint(endpoint);
146
147                return amazonClient;
148
149        }
150
151        private boolean isStackValid(final Stack stack) {
152                return stack != null;
153        }
154
155        private Stack waitForStackDelete() throws Exception {
156
157                final long timeStart = System.currentTimeMillis();
158
159                while (true) {
160
161                        if (isTimeoutPending(timeStart)) {
162                                return newStackWithStatus(StackStatus.DELETE_FAILED,
163                                                "stack delete timeout");
164                        }
165
166                        Stack stack = null;
167                        try {
168                                stack = getStack();
169                        } catch (final Exception e) {
170                                return newStackWithStatus(StackStatus.DELETE_FAILED,
171                                                e.toString());
172                        }
173
174                        if (!isStackValid(stack)) {
175                                return newStackWithStatus(StackStatus.DELETE_COMPLETE,
176                                                "stack delete invalid/missing");
177                        }
178
179                        final StackStatus status = StackStatus.fromValue(stack
180                                        .getStackStatus());
181
182                        switch (status) {
183                        case DELETE_IN_PROGRESS:
184                                final long timeCurrent = System.currentTimeMillis();
185                                final long timeDiff = timeCurrent - timeStart;
186                                logger.info("stack delete in progress; time=" + timeDiff / 1000);
187                                sleep();
188                                continue;
189                        case DELETE_COMPLETE:
190                                logger.info("stack delete complete");
191                                printStackEvents();
192                                return stack;
193                        default:
194                                logger.error("stack delete failed");
195                                return stack;
196                        }
197
198                }
199
200        }
201
202        private Stack newStackWithStatus(final StackStatus status,
203                        final String reason) {
204
205                final Stack stack = new Stack();
206
207                stack.setStackName(name);
208                stack.setStackStatus(status);
209                stack.setStackStatusReason(reason);
210
211                return stack;
212
213        }
214
215        private Stack waitForStackCreate() throws Exception {
216
217                final long timeStart = System.currentTimeMillis();
218
219                while (true) {
220
221                        if (isTimeoutPending(timeStart)) {
222                                return newStackWithStatus(StackStatus.CREATE_FAILED,
223                                                "stack create timeout");
224                        }
225
226                        Stack stack = null;
227                        try {
228                                stack = getStack();
229                        } catch (final Exception e) {
230                                return newStackWithStatus(StackStatus.CREATE_FAILED,
231                                                e.toString());
232                        }
233
234                        if (!isStackValid(stack)) {
235                                return newStackWithStatus(StackStatus.CREATE_FAILED,
236                                                "stack create invalid/missing");
237                        }
238
239                        final StackStatus status = StackStatus.fromValue(stack
240                                        .getStackStatus());
241
242                        switch (status) {
243                        case CREATE_IN_PROGRESS:
244                                final long timeCurrent = System.currentTimeMillis();
245                                final long timeDiff = timeCurrent - timeStart;
246                                logger.info("stack create in progress; time=" + timeDiff / 1000);
247                                sleep();
248                                continue;
249                        case CREATE_COMPLETE:
250                                logger.info("stack create success");
251                                printStackEvents();
252                                return stack;
253                        default:
254                                logger.error("stack create failure");
255                                return stack;
256                        }
257
258                }
259
260        }
261
262        private void printStackEvents() {
263
264                final DescribeStackEventsRequest request = new DescribeStackEventsRequest();
265
266                request.withStackName(name);
267
268                final DescribeStackEventsResult describeStackEvents = amazonClient
269                                .describeStackEvents(request);
270
271                final List<StackEvent> stackEvents = describeStackEvents
272                                .getStackEvents();
273
274                Collections.reverse(stackEvents);
275
276                logger.info("stack events:");
277
278                for (final StackEvent event : stackEvents) {
279
280                        final StringBuilder text = new StringBuilder(128);
281
282                        text.append("\n\t");
283                        text.append("time=");
284                        text.append(event.getTimestamp());
285
286                        text.append("\n\t");
287                        text.append("id=");
288                        text.append(event.getEventId());
289
290                        text.append("\n\t");
291                        text.append("type=");
292                        text.append(event.getResourceType());
293
294                        text.append("\n\t");
295                        text.append("status=");
296                        text.append(event.getResourceStatus());
297
298                        text.append("\n\t");
299                        text.append("reason=");
300                        text.append(event.getResourceStatusReason());
301
302                        logger.info("event {}", text);
303
304                }
305
306        }
307
308        private boolean isTimeoutPending(final long startTime) {
309                return (System.currentTimeMillis() - startTime) > (timeout * 1000);
310        }
311
312        private Stack getStack() throws Exception {
313
314                final DescribeStacksRequest request = new DescribeStacksRequest();
315
316                final DescribeStacksResult result = amazonClient
317                                .describeStacks(request);
318
319                for (final Stack stack : result.getStacks()) {
320                        if (name.equals(stack.getStackName())) {
321                                return stack;
322                        }
323                }
324
325                return null;
326
327        }
328
329        private void sleep() throws Exception {
330                try {
331                        Thread.sleep(waitBetweenAttempts * 1000);
332                } catch (final InterruptedException ie) {
333                        throw new IllegalStateException("operation interrupted; "
334                                        + "resources are left in inconsistent state; "
335                                        + "requires manual intervention");
336                }
337        }
