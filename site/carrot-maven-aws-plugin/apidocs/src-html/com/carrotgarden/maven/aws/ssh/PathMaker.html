008package com.carrotgarden.maven.aws.ssh;
009
010import java.io.File;
011import java.util.ArrayList;
012import java.util.List;
013
014import org.slf4j.Logger;
015
016public class PathMaker {
017
018        public static class Entry {
019
020                public final String source;
021                public final String target;
022
023                public Entry(final String source, final String target) {
024                        this.source = source;
025                        this.target = target;
026                }
027
028        }
029
030        //
031
032        private final Logger logger;
033
034        private final String source;
035        private final String target;
036
037        public PathMaker(final Logger logger, final String source,
038                        final String target) {
039
040                this.logger = logger;
041
042                this.source = source;
043                this.target = target;
044
045        }
046
047        private List<String> makeList(final String source) {
048
049                final File sourceDir = new File(source);
050
051                if (!sourceDir.exists()) {
052                        throw new IllegalArgumentException("source does not exist");
053                }
054
055                if (!sourceDir.isDirectory()) {
056                        throw new IllegalArgumentException("source is not a directory");
057                }
058
059                final File[] entryArray = sourceDir.listFiles();
060
061                final List<String> fileList = new ArrayList<String>();
062
063                for (final File entry : entryArray) {
064
065                        final String path = entry.getAbsolutePath();
066
067                        if (entry.isDirectory()) {
068                                fileList.addAll(makeList(path));
069                        } else {
070                                fileList.add(path);
071                        }
072
073                }
074
075                return fileList;
076
077        }
078
079        public List<Entry> getEntryList() {
080
081                final List<String> sourceList = makeList(source);
082
083                final List<Entry> entryList = new ArrayList<Entry>();
084
085                for (final String sourceFile : sourceList) {
086
087                        final String targetFile = sourceFile.replaceFirst(source, target);
088
089                        final Entry entry = new Entry(sourceFile, targetFile);
090
091                        entryList.add(entry);
092
093                        logger.debug("sourceFile={} targetFile={}", sourceFile, targetFile);
094
095                }
096
097                return entryList;
098
099        }
100
101}
