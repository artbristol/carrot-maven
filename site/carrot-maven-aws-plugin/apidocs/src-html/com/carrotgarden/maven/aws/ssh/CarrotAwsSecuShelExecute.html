008    package com.carrotgarden.maven.aws.ssh;
009    
010    /**
011     */
012    
013    import org.apache.maven.plugin.MojoFailureException;
014    
015    /**
016     * execute remote ssh commands
017     * 
018     * @goal secure-shell-execute
019     * 
020     * @phase prepare-package
021     * 
022     * @inheritByDefault true
023     * 
024     * @requiresDependencyResolution test
025     * 
026     */
027    public class CarrotAwsSecuShelExecute extends CarrotAwsSecuShel {
028    
029            /**
030             * ssh exec command
031             * 
032             * @required
033             * @parameter default-value="ls -las"
034             */
035            protected String sshCommand;
036    
037            /**
038             * {@inheritDoc}
039             */
040            @Override
041            public void execute() throws MojoFailureException {
042    
043                    try {
044    
045                            getLog().info("");
046    
047                            final SecureShell ssh = getSecureShell();
048    
049                            final int status = ssh.execute(sshCommand);
050    
051                            ensureStatusSuccess(status);
052    
053                    } catch (final Exception e) {
054    
055                            throw new MojoFailureException("command failed", e);
056                    }
057    
058            }
059    
060    }
