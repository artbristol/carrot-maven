008    package com.carrotgarden.maven.aws.ecc;
009    
010    import java.util.ArrayList;
011    import java.util.List;
012    
013    import org.slf4j.Logger;
014    
015    import com.amazonaws.auth.AWSCredentials;
016    import com.amazonaws.auth.BasicAWSCredentials;
017    import com.amazonaws.services.ec2.AmazonEC2;
018    import com.amazonaws.services.ec2.AmazonEC2Client;
019    import com.amazonaws.services.ec2.model.CreateImageRequest;
020    import com.amazonaws.services.ec2.model.CreateImageResult;
021    import com.amazonaws.services.ec2.model.DescribeImagesRequest;
022    import com.amazonaws.services.ec2.model.DescribeImagesResult;
023    import com.amazonaws.services.ec2.model.Image;
024    import com.amazonaws.services.ec2.model.StateReason;
025    
026    /**
027     * @author Andrei Pozolotin
028     */
029    public class ElasticCompute {
030    
031            private final Logger logger;
032    
033            private final String awsAccessKey;
034            private final String awsSecretKey;
035    
036            private final AmazonEC2 amazonClient;
037    
038            private final long timeout;
039            private final long waitBetweenAttempts;
040    
041            public ElasticCompute(final Logger logger, final long timeout,
042                            final String awsAccessKey, final String awsSecretKey) {
043    
044                    this.logger = logger;
045    
046                    this.timeout = timeout;
047    
048                    this.awsAccessKey = awsAccessKey;
049                    this.awsSecretKey = awsSecretKey;
050    
051                    this.amazonClient = getClient();
052    
053                    this.waitBetweenAttempts = 10;
054    
055            }
056    
057            private AmazonEC2 getClient() {
058    
059                    final AWSCredentials credentials = new BasicAWSCredentials(
060                                    this.awsAccessKey, this.awsSecretKey);
061    
062                    final AmazonEC2 amazonClient = new AmazonEC2Client(credentials);
063    
064                    return amazonClient;
065    
066            }
067    
068            public Image imageRegister(final String instanceId, final String name,
069                            final String description) throws Exception {
070    
071                    final CreateImageRequest request = new CreateImageRequest();
072    
073                    request.setInstanceId(instanceId);
074                    request.setName(name);
075                    request.setDescription(description);
076    
077                    final CreateImageResult result = amazonClient.createImage(request);
078    
079                    final String imageId = result.getImageId();
080    
081                    logger.info("register imageId=" + imageId);
082    
083                    final Image image = waitForImage(imageId);
084    
085                    return image;
086    
087            }
088    
089            public static enum State {
090    
091                    AVAILABLE("available"), //
092                    DEREGISTERED("deregistered"), //
093                    PENDING("pending"), //
094    
095                    UNKNOWN("unknown"), //
096    
097                    ;
098    
099                    public final String value;
100    
101                    State(final String value) {
102                            this.value = value;
103                    }
104    
105                    public static State fromValue(final String value) {
106                            for (final State known : values()) {
107                                    if (known.value.equals(value)) {
108                                            return known;
109                                    }
110                            }
111                            return UNKNOWN;
112                    }
113    
114            }
115    
116            private Image newImageWithStatus(final String state, final String code,
117                            final String message) {
118    
119                    final StateReason reason = new StateReason();
120                    reason.setCode(code);
121                    reason.setMessage(message);
122    
123                    final Image image = new Image();
124                    image.setState(state);
125                    image.setStateReason(reason);
126    
127                    return image;
128    
129            }
130    
131            private Image waitForImage(final String imageId) throws Exception {
132    
133                    final long timeStart = System.currentTimeMillis();
134    
135                    final List<String> imageIds = new ArrayList<String>();
136                    imageIds.add(imageId);
137    
138                    final DescribeImagesRequest request = new DescribeImagesRequest();
139                    request.setImageIds(imageIds);
140    
141                    while (true) {
142    
143                            final DescribeImagesResult result = amazonClient
144                                            .describeImages(request);
145    
146                            final List<Image> imageList = result.getImages();
147    
148                            final Image image;
149    
150                            if (isTimeoutPending(timeStart)) {
151                                    image = newImageWithStatus(State.UNKNOWN.value, "timeout",
152                                                    "image create timeout while waiting");
153                            } else if (imageList == null || imageList.isEmpty()) {
154                                    image = newImageWithStatus(State.UNKNOWN.value, "missing",
155                                                    "image create missing in descriptions");
156                            } else {
157                                    image = imageList.get(0);
158                            }
159    
160                            final String value = image.getState();
161    
162                            final State state = State.fromValue(value);
163    
164                            switch (state) {
165    
166                            case AVAILABLE:
167                                    logger.info("image create success");
168                                    return image;
169    
170                            case PENDING:
171                                    final long timeCurrent = System.currentTimeMillis();
172                                    final long timeDiff = timeCurrent - timeStart;
173                                    logger.info("image create in progress; time=" + timeDiff / 1000);
174                                    sleep();
175                                    break;
176    
177                            default:
178                                    logger.error("image create failure");
179                                    return image;
180    
181                            }
182    
183                    }
184    
185            }
186    
187            private void sleep() throws Exception {
188                    try {
189                            Thread.sleep(waitBetweenAttempts * 1000);
190                    } catch (final InterruptedException ie) {
191                            throw new IllegalStateException("operation interrupted; "
192                                            + "resources are left in inconsistent state; "
193                                            + "requires manual intervention");
194                    }
195            }
196    
197            private boolean isTimeoutPending(final long startTime) {
198                    return (System.currentTimeMillis() - startTime) > (timeout * 1000);
199            }
200    
201    }
