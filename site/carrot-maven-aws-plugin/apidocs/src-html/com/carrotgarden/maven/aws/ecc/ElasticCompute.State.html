016import com.amazonaws.auth.AWSCredentials;
017import com.amazonaws.services.ec2.AmazonEC2;
018import com.amazonaws.services.ec2.AmazonEC2Client;
019import com.amazonaws.services.ec2.model.CreateImageRequest;
020import com.amazonaws.services.ec2.model.CreateImageResult;

029/**
030 * @author Andrei Pozolotin
031 */
032public class ElasticCompute {
033
034        private final Logger logger;
035
036        private final AWSCredentials credentials;
037
038        private final AmazonEC2 amazonClient;
039
040        private final long timeout;
041        private final long waitBetweenAttempts;
042
043        private final String endpoint;
044
045        public ElasticCompute(final Logger logger, final long timeout,
046                        final AWSCredentials credentials, final String endpoint) {
047
048                this.logger = logger;
049
050                this.timeout = timeout;
051
052                this.credentials = credentials;
053
054                this.waitBetweenAttempts = 10;
055
056                this.endpoint = endpoint;
057
058                this.amazonClient = newClient(); // keep last
059
060        }
061
062        private AmazonEC2 newClient() {
063
064                final AmazonEC2 amazonClient = new AmazonEC2Client(credentials);
065
066                amazonClient.setEndpoint(endpoint);
067
068                return amazonClient;
069
070        }
071
072        public void tagCreate(final String resourceId, final String key,
073                        final String value) {
074
075                final CreateTagsRequest request = new CreateTagsRequest();
076
077                final Collection<String> resourceList = new ArrayList<String>(1);
078                resourceList.add(resourceId);
079
080                final Collection<Tag> tagList = new ArrayList<Tag>(1);
081                tagList.add(new Tag(key, value));
082
083                request.setResources(resourceList);
084                request.setTags(tagList);
085
086                logger.info("tag create request=" + request);
087
088                amazonClient.createTags(request);
089
090        }
091
092        public void tagDelete(final String resourceId, final String key,
093                        final String value) {
094
095                final DeleteTagsRequest request = new DeleteTagsRequest();
096
097                final Collection<String> resourceList = new ArrayList<String>(1);
098                resourceList.add(resourceId);
099
100                final Collection<Tag> tagList = new ArrayList<Tag>(1);
101                tagList.add(new Tag(key, value));
102
103                request.setResources(resourceList);
104                request.setTags(tagList);
105
106                logger.info("tag delete request=" + request);
107
108                amazonClient.deleteTags(request);
109
110        }
111
112        public Image imageRegister(final String instanceId, final String name,
113                        final String description) throws Exception {
114
115                final CreateImageRequest request = new CreateImageRequest();
116
117                request.setInstanceId(instanceId);
118                request.setName(name);
119                request.setDescription(description);
120
121                final CreateImageResult result = amazonClient.createImage(request);
122
123                final String imageId = result.getImageId();
124
125                logger.info("register imageId=" + imageId);
126
127                final Image image = waitForImage(imageId);
128
129                return image;
130
131        }
132
133        public static enum State {
134
135                AVAILABLE("available"), //
136                DEREGISTERED("deregistered"), //
137                PENDING("pending"), //
138
139                UNKNOWN("unknown"), //
140
141                ;
142
143                public final String value;
144
145                State(final String value) {
146                        this.value = value;
147                }
148
149                public static State fromValue(final String value) {
150                        for (final State known : values()) {
151                                if (known.value.equals(value)) {
152                                        return known;
153                                }
154                        }
155                        return UNKNOWN;
156                }
157
158        }
159
160        private Image newImageWithStatus(final String state, final String code,
161                        final String message) {
162
163                final StateReason reason = new StateReason();
164                reason.setCode(code);
165                reason.setMessage(message);
166
167                final Image image = new Image();
168                image.setState(state);
169                image.setStateReason(reason);
170
171                return image;
172
173        }
174
175        private Image waitForImage(final String imageId) throws Exception {
176
177                final long timeStart = System.currentTimeMillis();
178
179                final List<String> imageIds = new ArrayList<String>();
180                imageIds.add(imageId);
181
182                final DescribeImagesRequest request = new DescribeImagesRequest();
183                request.setImageIds(imageIds);
184
185                while (true) {
186
187                        final DescribeImagesResult result = amazonClient
188                                        .describeImages(request);
189
190                        final List<Image> imageList = result.getImages();
191
192                        final Image image;
193
194                        if (isTimeoutPending(timeStart)) {
195                                image = newImageWithStatus(State.UNKNOWN.value, "timeout",
196                                                "image create timeout while waiting");
197                        } else if (imageList == null || imageList.isEmpty()) {
198                                image = newImageWithStatus(State.UNKNOWN.value, "missing",
199                                                "image create missing in descriptions");
200                        } else {
201                                image = imageList.get(0);
202                        }
203
204                        final String value = image.getState();
205
206                        final State state = State.fromValue(value);
207
208                        switch (state) {
209
210                        case AVAILABLE:
211                                logger.info("image create success");
212                                return image;
213
214                        case PENDING:
215                                final long timeCurrent = System.currentTimeMillis();
216                                final long timeDiff = timeCurrent - timeStart;
217                                logger.info("image create in progress; time=" + timeDiff / 1000);
218                                sleep();
219                                break;
220
221                        default:
222                                logger.error("image create failure");
223                                return image;
224
225                        }
226
227                }
228
229        }
230
231        private void sleep() throws Exception {
232                try {
233                        Thread.sleep(waitBetweenAttempts * 1000);
234                } catch (final InterruptedException ie) {
235                        throw new IllegalStateException("operation interrupted; "
236                                        + "resources are left in inconsistent state; "
237                                        + "requires manual intervention");
238                }
239        }
240
241        private boolean isTimeoutPending(final long startTime) {
242                return (System.currentTimeMillis() - startTime) > (timeout * 1000);
243        }
