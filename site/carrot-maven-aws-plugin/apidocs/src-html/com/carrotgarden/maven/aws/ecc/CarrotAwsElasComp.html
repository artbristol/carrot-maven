008package com.carrotgarden.maven.aws.ecc;
009
010import org.apache.maven.settings.Server;
011import org.slf4j.Logger;
012
013import com.carrotgarden.maven.aws.CarrotAws;
014
015/**
016 * 
017 */
018public abstract class CarrotAwsElasComp extends CarrotAws {
019
020        /**
021         * AWS ElasticCompute
022         * 
023         * <a href=
024         * "http://docs.amazonwebservices.com/AWSSecurityCredentials/1.0/AboutAWSCredentials.html"
025         * >amazon security credentials</a>
026         * 
027         * stored in
028         * 
029         * <a href=
030         * "http://www.sonatype.com/books/mvnref-book/reference/appendix-settings-sect-details.html"
031         * >maven settings.xml</a>
032         * 
033         * under server id entry; username="Access Key ID",
034         * password="Secret Access Key";
035         * 
036         * @required
037         * @parameter default-value="com.example.aws.compute"
038         */
039        protected String computeServerId;
040
041        /**
042         * AWS ElasticCompute operation timeout; seconds
043         * 
044         * @required
045         * @parameter default-value="600"
046         */
047        protected Long computeTimeout;
048
049        /**
050         * AWS ElasticCompute
051         * 
052         * <a href=
053         * "http://docs.amazonwebservices.com/general/latest/gr/rande.html#ec2_region"
054         * >optional api end point url</a>
055         * 
056         * which controls amazon region selection;
057         * 
058         * when omitted, will be constructed from {@link #amazonRegion}
059         * 
060         * @parameter
061         */
062        protected String computeEndpoint;
063
064        protected String getComputeEndpoint() {
065                if (computeEndpoint == null) {
066                        return "https://ec2." + amazonRegion + ".amazonaws.com";
067                } else {
068                        return computeEndpoint;
069                }
070        }
071
072        protected ElasticCompute getElasticCompute() throws Exception {
073
074                final Server server = settings.getServer(computeServerId);
075
076                if (server == null) {
077                        throw new IllegalArgumentException(
078                                        "server definition is missing for serverId="
079                                                        + computeServerId);
080                }
081
082                final String username = server.getUsername();
083                final String password = server.getPassword();
084
085                final Logger logger = getLogger(ElasticCompute.class);
086
087                final ElasticCompute compute = new ElasticCompute(logger,
088                                computeTimeout, username, password, getComputeEndpoint());
089
090                return compute;
091
092        }
093
094}
