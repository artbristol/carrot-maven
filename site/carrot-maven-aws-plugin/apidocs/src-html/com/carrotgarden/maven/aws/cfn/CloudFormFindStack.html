008package com.carrotgarden.maven.aws.cfn;
009
010import java.util.Properties;
011
012import org.apache.maven.plugin.MojoFailureException;
013
014import com.amazonaws.services.cloudformation.model.Stack;
015
016/**
017 * cloud formation:
018 * 
019 * <b><a href=
020 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/APIReference/API_DescribeStacks.html"
021 * >find stack</a></b>
022 * 
023 * by name
024 * 
025 * @goal cloud-formation-find-stack
026 * 
027 * @phase prepare-package
028 * 
029 * @inheritByDefault true
030 * 
031 * @requiresDependencyResolution test
032 * 
033 */
034public class CloudFormFindStack extends CloudForm {
035
036        /**
037         * name of project.property which will contain {@link Stack} instance after
038         * execution of this maven goal, which can be used by groovy script as
039         * follows:
040         * 
041         * <pre>
042         * def stack = project.properties["amazonStack"]
043         * println "name = " + stack.stackName
044         * println "status = " + stack.stackStatus
045         * </pre>
046         * 
047         * @required
048         * @parameter default-value="amazonStack"
049         */
050        private String stackResultProperty;
051
052        /**
053         * {@inheritDoc}
054         */
055        @Override
056        public void execute() throws MojoFailureException {
057
058                try {
059
060                        getLog().info("stack find init [" + stackName() + "]");
061
062                        final CarrotCloudForm formation = newCloudFormation(null, null);
063
064                        final Stack stack = formation.findStack();
065
066                        getLog().info("stack find result : " + stack);
067
068                        final Properties props = project().getProperties();
069
070                        if (stack == null) {
071                                props.remove(stackResultProperty);
072                        } else {
073                                props.put(stackResultProperty, stack);
074                        }
075
076                        getLog().info("stack find done [" + stackName() + "]");
077
078                } catch (final Exception e) {
079
080                        throw new MojoFailureException("bada-boom", e);
081
082                }
083
084        }
085
086}
