008package com.carrotgarden.maven.aws.cfn;
009
010import java.io.File;
011import java.util.HashMap;
012import java.util.Map;
013import java.util.Properties;
014
015import org.apache.commons.io.FileUtils;
016import org.apache.maven.settings.Server;
017import org.slf4j.Logger;
018
019import com.carrotgarden.maven.aws.CarrotAws;
020
021/**
022 * 
023 */
024public abstract class CarrotAwsCloudForm extends CarrotAws {
025
026        /**
027         * AWS CloudFormation stack name; must be unique under your aws account /
028         * region
029         * 
030         * @required
031         * @parameter default-value="amazon-builder"
032         */
033        protected String stackName;
034
035        /**
036         * AWS CloudFormation
037         * 
038         * <a href=
039         * "http://docs.amazonwebservices.com/AWSSecurityCredentials/1.0/AboutAWSCredentials.html"
040         * >amazon security credentials</a>
041         * 
042         * stored in
043         * 
044         * <a href=
045         * "http://www.sonatype.com/books/mvnref-book/reference/appendix-settings-sect-details.html"
046         * >maven settings.xml</a>
047         * 
048         * under server id entry; username="Access Key ID",
049         * password="Secret Access Key";
050         * 
051         * @required
052         * @parameter default-value="com.example.aws.stack"
053         */
054        protected String stackServerId;
055
056        /**
057         * AWS CloudFormation operation timeout; seconds
058         * 
059         * @required
060         * @parameter default-value="600"
061         */
062        protected Long stackTimeout;
063
064        /**
065         * AWS CloudFormation
066         * 
067         * <a href=
068         * "http://docs.amazonwebservices.com/general/latest/gr/rande.html#cfn_region"
069         * >optional api end point url</a>
070         * 
071         * which controls amazon region selection;
072         * 
073         * when omitted, will be constructed from {@link #amazonRegion}
074         * 
075         * @parameter
076         */
077        protected String stackEndpoint;
078
079        protected String getStackEndpoint() {
080                if (stackEndpoint == null) {
081                        return "https://cloudformation." + amazonRegion + ".amazonaws.com";
082                } else {
083                        return stackEndpoint;
084                }
085        }
086
087        //
088
089        protected CloudFormation getCloudFormation(final File templateFile,
090                        final Properties inputProps, final Map<String, String> inputParams)
091                        throws Exception {
092
093                /** merge template parameters */
094
095                final Map<String, String> stackParams = new HashMap<String, String>();
096
097                /** from properties file */
098                stackParams.putAll(safeMap(inputProps));
099
100                /** from maven pom.xml */
101                stackParams.putAll(safeMap(inputParams));
102
103                /** */
104
105                final String stackTemplate = safeTemplate(templateFile);
106
107                /** */
108
109                final Server server = settings.getServer(stackServerId);
110
111                if (server == null) {
112                        throw new IllegalArgumentException(
113                                        "settings.xml : server definition is missing for serverId="
114                                                        + stackServerId);
115                }
116
117                final String username = server.getUsername();
118                final String password = server.getPassword();
119
120                /** */
121
122                final Logger logger = getLogger(CloudFormation.class);
123
124                /** */
125
126                final CloudFormation formation = new CloudFormation(logger, stackName,
127                                stackTemplate, stackParams, stackTimeout, username, password,
128                                getStackEndpoint());
129
130                return formation;
131
132        }
133
134        protected String safeTemplate(final File templateFile) throws Exception {
135                if (templateFile == null || !templateFile.exists()) {
136                        return "{}";
137                } else {
138                        return FileUtils.readFileToString(templateFile);
139                }
140        }
141
142}
