008package com.carrotgarden.maven.aws.cfn;
009
010import org.apache.maven.plugin.MojoFailureException;
011
012import com.amazonaws.services.cloudformation.model.Stack;
013import com.amazonaws.services.cloudformation.model.StackStatus;
014
015/**
016 * cloud formation:
017 * 
018 * <b><a href=
019 * "http://docs.amazonwebservices.com/AWSCloudFormation/latest/APIReference/API_DeleteStack.html"
020 * >delete stack</a></b>
021 * 
022 * based on:
023 * 
024 * <b>stack name</b>;
025 * 
026 * ; wait for completion or fail ({@link #stackTimeout})
027 * 
028 * @goal cloud-formation-delete
029 * 
030 * @phase prepare-package
031 * 
032 * @inheritByDefault true
033 * 
034 * @requiresDependencyResolution test
035 * 
036 */
037public class CarrotAwsCloudFormDelete extends CarrotAwsCloudForm {
038
039        /**
040         * {@inheritDoc}
041         */
042        @Override
043        public void execute() throws MojoFailureException {
044
045                try {
046
047                        getLog().info("stack delete init [" + stackName() + "]");
048
049                        final CloudFormation formation = newCloudFormation(null, null);
050
051                        final Stack stack = formation.stackDelete();
052
053                        final StackStatus status = StackStatus.fromValue(stack
054                                        .getStackStatus());
055
056                        switch (status) {
057                        case DELETE_COMPLETE:
058                                break;
059                        default:
060                                throw new IllegalStateException("stack delete failed");
061                        }
062
063                        getLog().info("stack delete stack=\n" + stack);
064
065                        getLog().info("stack delete done [" + stackName() + "]");
066
067                } catch (final Exception e) {
068
069                        throw new MojoFailureException("bada-boom", e);
070
071                }
072
073        }
074
075}
