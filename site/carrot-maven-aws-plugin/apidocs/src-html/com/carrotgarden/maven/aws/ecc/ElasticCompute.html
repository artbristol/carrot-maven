008package com.carrotgarden.maven.aws.ecc;
009
010import java.util.ArrayList;
011import java.util.List;
012
013import org.slf4j.Logger;
014
015import com.amazonaws.auth.AWSCredentials;
016import com.amazonaws.auth.BasicAWSCredentials;
017import com.amazonaws.services.ec2.AmazonEC2;
018import com.amazonaws.services.ec2.AmazonEC2Client;
019import com.amazonaws.services.ec2.model.CreateImageRequest;
020import com.amazonaws.services.ec2.model.CreateImageResult;
021import com.amazonaws.services.ec2.model.DescribeImagesRequest;
022import com.amazonaws.services.ec2.model.DescribeImagesResult;
023import com.amazonaws.services.ec2.model.Image;
024import com.amazonaws.services.ec2.model.StateReason;
025
026/**
027 * @author Andrei Pozolotin
028 */
029public class ElasticCompute {
030
031        private final Logger logger;
032
033        private final String awsAccessKey;
034        private final String awsSecretKey;
035
036        private final AmazonEC2 amazonClient;
037
038        private final long timeout;
039        private final long waitBetweenAttempts;
040
041        private final String endpoint;
042
043        public ElasticCompute(final Logger logger, final long timeout,
044                        final String awsAccessKey, final String awsSecretKey,
045                        final String endpoint) {
046
047                this.logger = logger;
048
049                this.timeout = timeout;
050
051                this.awsAccessKey = awsAccessKey;
052                this.awsSecretKey = awsSecretKey;
053
054                this.waitBetweenAttempts = 10;
055
056                this.endpoint = endpoint;
057
058                this.amazonClient = newClient(); // keep last
059
060        }
061
062        private AmazonEC2 newClient() {
063
064                final AWSCredentials credentials = new BasicAWSCredentials(
065                                this.awsAccessKey, this.awsSecretKey);
066
067                final AmazonEC2 amazonClient = new AmazonEC2Client(credentials);
068
069                amazonClient.setEndpoint(endpoint);
070
071                return amazonClient;
072
073        }
074
075        public Image imageRegister(final String instanceId, final String name,
076                        final String description) throws Exception {
077
078                final CreateImageRequest request = new CreateImageRequest();
079
080                request.setInstanceId(instanceId);
081                request.setName(name);
082                request.setDescription(description);
083
084                final CreateImageResult result = amazonClient.createImage(request);
085
086                final String imageId = result.getImageId();
087
088                logger.info("register imageId=" + imageId);
089
090                final Image image = waitForImage(imageId);
091
092                return image;
093
094        }
095
096        public static enum State {
097
098                AVAILABLE("available"), //
099                DEREGISTERED("deregistered"), //
100                PENDING("pending"), //
101
102                UNKNOWN("unknown"), //
103
104                ;
105
106                public final String value;
107
108                State(final String value) {
109                        this.value = value;
110                }
111
112                public static State fromValue(final String value) {
113                        for (final State known : values()) {
114                                if (known.value.equals(value)) {
115                                        return known;
116                                }
117                        }
118                        return UNKNOWN;
119                }
120
121        }
122
123        private Image newImageWithStatus(final String state, final String code,
124                        final String message) {
125
126                final StateReason reason = new StateReason();
127                reason.setCode(code);
128                reason.setMessage(message);
129
130                final Image image = new Image();
131                image.setState(state);
132                image.setStateReason(reason);
133
134                return image;
135
136        }
137
138        private Image waitForImage(final String imageId) throws Exception {
139
140                final long timeStart = System.currentTimeMillis();
141
142                final List<String> imageIds = new ArrayList<String>();
143                imageIds.add(imageId);
144
145                final DescribeImagesRequest request = new DescribeImagesRequest();
146                request.setImageIds(imageIds);
147
148                while (true) {
149
150                        final DescribeImagesResult result = amazonClient
151                                        .describeImages(request);
152
153                        final List<Image> imageList = result.getImages();
154
155                        final Image image;
156
157                        if (isTimeoutPending(timeStart)) {
158                                image = newImageWithStatus(State.UNKNOWN.value, "timeout",
159                                                "image create timeout while waiting");
160                        } else if (imageList == null || imageList.isEmpty()) {
161                                image = newImageWithStatus(State.UNKNOWN.value, "missing",
162                                                "image create missing in descriptions");
163                        } else {
164                                image = imageList.get(0);
165                        }
166
167                        final String value = image.getState();
168
169                        final State state = State.fromValue(value);
170
171                        switch (state) {
172
173                        case AVAILABLE:
174                                logger.info("image create success");
175                                return image;
176
177                        case PENDING:
178                                final long timeCurrent = System.currentTimeMillis();
179                                final long timeDiff = timeCurrent - timeStart;
180                                logger.info("image create in progress; time=" + timeDiff / 1000);
181                                sleep();
182                                break;
183
184                        default:
185                                logger.error("image create failure");
186                                return image;
187
188                        }
189
190                }
191
192        }
193
194        private void sleep() throws Exception {
195                try {
196                        Thread.sleep(waitBetweenAttempts * 1000);
197                } catch (final InterruptedException ie) {
198                        throw new IllegalStateException("operation interrupted; "
199                                        + "resources are left in inconsistent state; "
200                                        + "requires manual intervention");
201                }
202        }
203
204        private boolean isTimeoutPending(final long startTime) {
205                return (System.currentTimeMillis() - startTime) > (timeout * 1000);
206        }
207
208}
