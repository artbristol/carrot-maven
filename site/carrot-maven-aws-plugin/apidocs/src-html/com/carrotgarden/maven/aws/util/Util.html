030        public static String concatenate(final List<String> list,
031                        final String separator) {
032
033                final StringBuilder text = new StringBuilder();
034
035                for (final String item : list) {
036                        text.append(item);
037                        text.append(separator);
038                }
039
040                return text.toString();
041
042        }
043
044        public static Field findField(final Class<?> klaz, final String fieldName) {
045
046                Class<?> type = klaz;
047
048                do {
049
050                        for (final Field field : type.getDeclaredFields()) {
051
052                                final String name = field.getName();
053
054                                if (!name.equals(fieldName)) {
055                                        continue;
056                                }
057
058                                field.setAccessible(true);
059
060                                return field;
061
062                        }
063
064                        type = type.getSuperclass();
065
066                } while (klaz != null);
067
068                return null;
069
070        }
071
072        public static boolean isValidFolder(final File file) {
073
074                if (file == null) {
075                        return false;
076                }
077
078                if (!file.exists()) {
079                        return false;
080                }
081
082                if (!file.isDirectory()) {
083                        return false;
084                }
085
086                if (!file.canRead()) {
087                        return false;
088                }
089
090                if (!file.canWrite()) {
091                        return false;
092                }
093
094                return true;
095
096        }

107        public static void overrideInstanceProps(final Log log,
108                        final Object instance, final String prefix,
109                        final Map<String, String> props) {
110
111                final Set<Map.Entry<String, String>> entrySet = //
112                new HashSet<Entry<String, String>>(props.entrySet());
113
114                final Class<?> klaz = instance.getClass();
115
116                for (final Map.Entry<String, String> entry : entrySet) {
117
118                        final String key = entry.getKey();
119                        final String value = entry.getValue();
120
121                        if (!key.startsWith(prefix)) {
122                                continue;
123                        }
124
125                        props.remove(key);
126
127                        try {
128
129                                final Field field = findField(klaz, key);
130
131                                field.set(instance, value);
132
133                                log.info("override : " + key + "=" + value);
134
135                        } catch (final Exception e) {
136
137                                log.warn("override : invalid stack param=" + key, e);
138
139                        }
140
141                }
142
143        }
144
145        public static Properties propsLoad(final Log log, final File file)
146                        throws Exception {
147
148                final Properties props = new Properties();
149
150                if (file == null || !file.exists()) {
151                        log.debug("file == null || !file.exists()");
152                        return props;
153                }
154
155                final Reader reader = new FileReader(file);
156
157                props.load(reader);
158
159                reader.close();
160
161                return props;
162
163        }
164
165        public static void propsReport(final Log log, final String title,
166                        final Properties props) {
167
168                log.info("properties : " + title);
169
170                final Object[] keyArray = props.keySet().toArray();
171
172                Arrays.sort(keyArray);
173
174                for (final Object key : keyArray) {
175                        log.info("\t" + key + "=" + props.get(key));
176                }
177
178        }
179
180        public static void propsSave(final Log log, final Properties props,
181                        final File file) throws Exception {
182
183                if (props == null || file == null) {
184                        log.debug("props == null || file == null");
185                        return;
186                }
187
188                final File folder = file.getParentFile();
189
190                if (!folder.exists()) {
191                        folder.mkdirs();
192                }
193
194                final Writer writer = new FileWriter(file);
195
196                props.store(writer, null);
197
198        }

200        public static Map<String, String> safeMap(final Map<String, String> props) {
201
202                final Map<String, String> map = new TreeMap<String, String>();
203
204                if (props == null) {
205                        return map;
206                }
207
208                return props;
209
210        }
211
212        public static Map<String, String> safeMap(final Properties props) {
213
214                final Map<String, String> map = new TreeMap<String, String>();
215
216                if (props == null) {
217                        return map;
218                }
219
220                for (final Map.Entry<Object, Object> entry : props.entrySet()) {
221
222                        final String key = entry.getKey().toString();
223                        final String value = entry.getValue().toString();
224
225                        map.put(key, value);
226
227                }
228
229                return map;
230
231        }

233        public static long safeNumber(final Log log, final String numberText,
234                        final long numberDefault) {
235                try {
236                        return Long.parseLong(numberText);
237                } catch (final Throwable e) {
238                        log.warn("using numberDefault=" + numberDefault);
239                        return numberDefault;
240                }
241        }

249                        throw new IllegalStateException(message);

251                }
